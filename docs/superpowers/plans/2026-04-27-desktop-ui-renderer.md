# Desktop UI Renderer Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the first OpenBBQ desktop renderer slice for the approved warm editorial subtitle workflow UI.

**Architecture:** Add a `desktop/` React application with a typed local API boundary, mockable data adapters, and screen-level components for source import, workflow arrangement, task monitoring, and results review. Keep backend gap closure out of this plan; expose those unsupported capabilities through typed adapter seams so later backend plans can replace mocks with real API calls.

**Tech Stack:** Electron-ready React + TypeScript + Vite, pnpm, Tailwind CSS as the only CSS strategy, lucide-react icons, Vitest, Testing Library, Playwright.

---

## Scope

This plan implements a renderer-first desktop UI slice. It produces working,
testable UI screens and state transitions that match
`docs/superpowers/specs/2026-04-27-desktop-ui-design.md`.

This plan does not implement:

- Electron main/preload sidecar lifecycle;
- backend workflow template persistence;
- run-scoped checkpoint retry API;
- editable segment persistence API;
- server-derived waveform data;
- packaging or installers.

Those backend and shell capabilities need separate implementation plans. This
plan still creates interfaces for them so the UI code does not need a rewrite
later.

```ts
export function formatTimestamp(ms: number): string {
  const totalSeconds = Math.floor(ms / 1000);
  const minutes = Math.floor(totalSeconds / 60);
  const seconds = totalSeconds % 60;
  const millis = ms % 1000;
  return `${minutes.toString().padStart(2, "0")}:${seconds.toString().padStart(2, "0")}.${millis
    .toString()
    .padStart(3, "0")}`;
}

export function formatRange(startMs: number, endMs: number): string {
  return `${formatTimestamp(startMs)} -> ${formatTimestamp(endMs)}`;
}
```

```tsx
import { clsx } from "clsx";

type ToggleProps = {
  checked: boolean;
  disabled?: boolean;
  label: string;
  onChange?: (checked: boolean) => void;
};

export function Toggle({ checked, disabled = false, label, onChange }: ToggleProps) {
  return (
    <button
      type="button"
      aria-label={label}
      aria-pressed={checked}
      disabled={disabled}
      onClick={() => {
        if (!disabled) onChange?.(!checked);
      }}
      className={clsx(
        "relative h-6 w-11 rounded-full transition-transform duration-150 active:scale-95 focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-accent",
        disabled ? "cursor-not-allowed bg-[#c9b79c]" : checked ? "bg-accent" : "bg-[#d8c8b1]"
      )}
    >
      <span
        className={clsx(
          "absolute top-1 h-4 w-4 rounded-full bg-[#fff8ea] transition-transform duration-150",
          checked ? "translate-x-6" : "translate-x-1"
        )}

  );
}
```

```tsx
import { useMemo, useState } from "react";
import type { StepParameter, WorkflowStep } from "../lib/types";

import { Toggle } from "./Toggle";

type WorkflowEditorProps = {
  initialSteps: WorkflowStep[];
  onContinue: (steps: WorkflowStep[]) => void;
};

function isLocked(step: WorkflowStep) {
  return step.status === "locked";
}

function ParameterField({ parameter }: { parameter: StepParameter }) {
  if (parameter.kind === "toggle") {
    return (
      <div className="flex items-center justify-between rounded-md bg-paper-muted p-3 shadow-control">
        <div>
          <div className="font-extrabold">{parameter.label}</div>
          <div className="mt-0.5 text-xs text-muted">{parameter.description}</div>
        </div>
        <Toggle checked={parameter.value} label={parameter.label} />
      </div>
    );
  }

  return (
    <label className="grid gap-2 text-xs text-muted">
      {parameter.label}
      <input
        value={parameter.value}
        readOnly
        className="min-h-10 rounded-md bg-paper-muted px-3 text-sm text-ink shadow-control focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-accent"
      />
    </label>
  );
}

export function WorkflowEditor({ initialSteps, onContinue }: WorkflowEditorProps) {
  const [steps, setSteps] = useState(initialSteps);
  const [selectedStepId, setSelectedStepId] = useState(
    initialSteps.find((step) => step.selected)?.id ?? initialSteps[0]?.id
  );
  const selectedStep = useMemo(
    () => steps.find((step) => step.id === selectedStepId) ?? steps[0],
    [selectedStepId, steps]
  );

  function toggleStep(stepId: string) {
    setSteps((current) =>
      current.map((step) => {
        if (step.id !== stepId || isLocked(step)) return step;
        return { ...step, status: step.status === "enabled" ? "disabled" : "enabled" };
      })
    );
  }

  return (
    <main className="grid grid-rows-[auto_1fr_auto] gap-4">
      <header className="flex items-end justify-between">
        <div>
          <p className="text-[11px] uppercase text-muted">New task - Step 2 of 4</p>
          <h1 className="mt-2 font-serif text-[39px] leading-none">Arrange workflow</h1>
        </div>
        <div className="rounded-full bg-[#e9deca] px-3 py-2 text-xs text-muted">
          Source -> Workflow -> Settings -> Preflight
        </div>
      </header>

      <section className="grid grid-cols-[minmax(430px,1fr)_minmax(360px,0.82fr)] gap-[18px]">
        <div className="rounded-xl bg-paper p-5 shadow-panel">
          <div className="flex items-start justify-between gap-4">
            <div>
              <p className="text-xs uppercase text-muted">Workflow template</p>
              <h2 className="mt-2 text-[22px] font-extrabold">Local video -&gt; translated SRT</h2>
              <p className="mt-1.5 text-sm text-muted">Toggle optional steps and edit selected step parameters.</p>
            </div>
            <span className="rounded-full bg-paper-selected px-2.5 py-1.5 text-xs text-[#8c4d29]">Built-in</span>
          </div>

          <div className="mt-[18px] grid gap-2.5">
            {steps.map((step, index) => {
              const selected = step.id === selectedStep.id;
              const locked = isLocked(step);
              return (
                <button
                  key={step.id}
                  type="button"
                  onClick={() => setSelectedStepId(step.id)}
                  className={[
                    "grid grid-cols-[34px_1fr_86px] items-center gap-3 rounded-lg p-3 text-left transition-transform duration-150 active:scale-[0.99] focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-accent",
                    selected ? "bg-paper-selected shadow-selected" : "bg-paper-muted shadow-control"
                  ].join(" ")}
                >
                  <span className={selected ? "flex h-6 w-6 items-center justify-center rounded-full bg-accent text-xs text-[#fff8ea]" : "flex h-6 w-6 items-center justify-center rounded-full bg-ink-brown text-xs text-[#fff8ea]"}>
                    {index + 1}
                  </span>
                  <span>
                    <strong>{step.name}</strong>
                    <br />
                    <span className="text-xs text-muted">{step.summary}</span>
                  </span>
                  <span className="flex items-center justify-end gap-1.5">
                    <span className="text-[11px] text-muted">{locked ? "locked" : step.status === "enabled" ? "on" : "off"}</span>
                    <span onClick={(event) => event.stopPropagation()}>
                      <Toggle
                        checked={step.status !== "disabled"}
                        disabled={locked}
                        label={`Toggle ${step.name}`}
                        onChange={() => toggleStep(step.id)}
                      />
                    </span>
                  </span>
                </button>
              );
            })}
          </div>
        </div>

        <aside className="rounded-xl bg-paper p-5 shadow-panel">
          <p className="text-xs uppercase text-muted">Selected step parameters</p>
```

```tsx
import { useEffect, useMemo, useState } from "react";
import { formatRange, formatTimestamp } from "../lib/format";
import type { ReviewModel, Segment } from "../lib/types";
import { Button } from "./Button";
import { Waveform } from "./Waveform";

type ResultsReviewProps = {
  model: ReviewModel;
  onSegmentChange: (segment: Segment) => Promise<void>;
};

export function ResultsReview({ model, onSegmentChange }: ResultsReviewProps) {
  const [segments, setSegments] = useState(model.segments);
  const [activeSegmentId, setActiveSegmentId] = useState(model.activeSegmentId);
  const activeSegment = useMemo(
    () => segments.find((segment) => segment.id === activeSegmentId) ?? segments[0],
    [activeSegmentId, segments]
  );
  const saveState = segments.some((segment) => segment.savedState === "saving")
    ? "Saving"
    : segments.some((segment) => segment.savedState === "error")
      ? "Save error"
      : "Saved";

  function updateSegment(segmentId: string, patch: Partial<Pick<Segment, "transcript" | "translation">>) {
    setSegments((current) =>
      current.map((segment) =>
        segment.id === segmentId ? { ...segment, ...patch, savedState: "saving" } : segment
      )
    );
  }

  useEffect(() => {
    const saving = segments.filter((segment) => segment.savedState === "saving");
    if (saving.length === 0) return;
    const timeout = window.setTimeout(() => {
      for (const segment of saving) {
        void onSegmentChange(segment).then(() => {
          setSegments((current) =>
            current.map((item) => (item.id === segment.id ? { ...item, savedState: "saved" } : item))
          );
        });
      }
    }, 350);
    return () => window.clearTimeout(timeout);
  }, [onSegmentChange, segments]);

  return (
    <main className="grid grid-rows-[auto_1fr] gap-4">
      <header className="flex items-end justify-between">
        <div>
          <p className="text-[11px] uppercase text-muted">Review results</p>
          <h1 className="mt-2 font-serif text-[38px] leading-none">{model.title}</h1>
          <p className="mt-1.5 text-sm text-muted">Completed - edits autosave</p>
        </div>
        <div className="flex items-center gap-2">
          <span className="text-xs text-ready">{saveState}</span>
          <Button variant="primary">Export SRT</Button>
        </div>
      </header>

      <section className="grid min-h-[560px] grid-cols-[minmax(500px,1.08fr)_minmax(380px,0.92fr)] gap-[18px]">
        <div className="grid content-start gap-3.5">
          <div className="rounded-xl bg-paper p-4 shadow-panel">
            <div className="relative aspect-video overflow-hidden rounded-lg bg-log-bg shadow-inner">
              <div className="absolute left-5 top-[18px] text-xs text-[#d7c4a7]">
                {formatTimestamp(model.currentMs)} / {formatTimestamp(model.durationMs)}
              </div>
              <div className="absolute inset-x-16 bottom-8 rounded-md bg-black/70 px-3 py-2 text-center text-[15px] leading-snug text-[#fff8ea]">
                {activeSegment?.translation}
              </div>
              <button
                type="button"
                aria-label="Play preview"
                className="absolute bottom-[18px] left-[18px] flex h-9 w-9 items-center justify-center rounded-full bg-[#fff8ea] text-log-bg"
              >
                Play
              </button>
            </div>
          </div>

          <div className="rounded-xl bg-paper px-4 py-3.5 shadow-panel">
            <div className="mb-2.5 flex items-center justify-between">
              <div>
                <p className="text-xs uppercase text-muted">Audio loudness</p>
                <p className="mt-1 text-xs text-muted">Waveform segments match the editable cards.</p>
              </div>
              <span className="text-xs text-[#8c4d29]">Segment {activeSegment?.index}</span>
            </div>
            <Waveform model={{ ...model, segments, activeSegmentId }} onSelectSegment={setActiveSegmentId} />
          </div>
        </div>

        <aside className="grid min-h-[560px] grid-rows-[auto_1fr] rounded-xl bg-paper p-4 shadow-panel">
          <div className="mb-3 flex items-center justify-between">
            <div>
              <p className="text-xs uppercase text-muted">Editable segments</p>
              <p className="mt-1 text-sm text-muted">Transcript and translation autosave.</p>
            </div>
            <span className="text-xs text-ready">{saveState}</span>
          </div>

          <div className="grid content-start gap-2.5 overflow-hidden">
            {segments.map((segment) => {
              const active = segment.id === activeSegmentId;
              return (
                <article
                  key={segment.id}
                  className={active ? "rounded-lg bg-paper-selected p-3 shadow-selected" : "rounded-lg bg-paper-muted p-3 shadow-control"}
                  onFocus={() => setActiveSegmentId(segment.id)}
                >
                  <div className={active ? "flex justify-between text-xs text-[#8c4d29]" : "flex justify-between text-xs text-[#8c7b61]"}>
                    <span>{formatRange(segment.startMs, segment.endMs)}</span>
                    <span>Segment {segment.index}</span>
                  </div>

                  <label className="mt-2 grid gap-1 text-[11px] text-[#8c7b61]">
                    Transcript
                    <textarea
                      aria-label={`Transcript for segment ${segment.index}`}
                      value={segment.transcript}
                      onChange={(event) => updateSegment(segment.id, { transcript: event.target.value })}
                      className="min-h-[58px] resize-none rounded-md bg-paper p-2 text-sm leading-snug text-ink shadow-control focus-visible:outline focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-accent"
```
